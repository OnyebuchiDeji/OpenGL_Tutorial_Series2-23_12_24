#version 330 core

layout (location=0) in vec3 vertexPos;
layout (location=1) in vec2 vertexTexCoord;
layout (location=2) in vec3 vertexNormal;

out vec3 fragmentTexCoord;
out vec3 fragmentNormal;

uniform mat4 u_Model;
uniform mat4 u_View;
uniform float u_ObjectType;

void main()
{
    gl_Position = u_View * u_Model * vec4(vertexPos, 1.0);
    fragmentTexCoord = vec3(vertexTexCoord, u_ObjectType);
    fragmentNormal = normalize(mat3(u_Model) * vertexNormal);
}
